# API Key Manager
As simple as it gets


## Basic Usage

    import { ApiKeyManager } from "./api-key-manager"

    const apiKeyManager = ApiKeyManager.getInstance()

    const apiKey: string = apiKeyManager.generateNewAPIKey()

    console.lot(apiKey)
